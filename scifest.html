// Include necessary libraries
#include  // For controlling motors
#include  // For detecting color

// Define pins for motors
#define MOTOR1_PIN 3
#define MOTOR2_PIN 5

// Define pins for color sensor
#define COLOR_SENSOR_PIN A0

// Define pin for buzzer
#define BUZZER_PIN 7

// Define threshold for black color
#define BLACK_THRESHOLD 500

// Define motor speeds
#define MOTOR_SPEED 200 // Speed for forward and backward movement
#define TURN_SPEED 150 // Speed for turning

// Create instances for motors and color sensor
Servo motor1;
Servo motor2;
ColorSensor colorSensor;

// Setup function
void setup() {
  // Initialize motors
  motor1.attach(MOTOR1_PIN);
  motor2.attach(MOTOR2_PIN);

  // Initialize color sensor
  colorSensor.begin(COLOR_SENSOR_PIN);

  // Initialize buzzer
  pinMode(BUZZER_PIN, OUTPUT);
}

// Main loop
void loop() {
  // Read color sensor value
  int colorValue = colorSensor.read();

  // Check if color is black
  if (colorValue < BLACK_THRESHOLD) {
    // Activate buzzer for 5 seconds
    digitalWrite(BUZZER_PIN, HIGH);
    delay(5000);
    digitalWrite(BUZZER_PIN, LOW);

    // Move forward with one wheel while moving backward with the other
    motor1.write(MOTOR_SPEED);
    motor2.write(-MOTOR_SPEED);
    delay(500);

    // Briefly move forward
    motor1.write(MOTOR_SPEED);
    motor2.write(MOTOR_SPEED);
    delay(500);

    // Return to original direction
    motor1.write(MOTOR_SPEED);
    motor2.write(-MOTOR_SPEED);
    delay(500);

    // Continue forward
    motor1.write(MOTOR_SPEED);
    motor2.write(MOTOR_SPEED);
    delay(1000);

    // Turn to opposite side
    motor1.write(TURN_SPEED);
    motor2.write(-TURN_SPEED);
    delay(1000);

    // Advance
    motor1.write(MOTOR_SPEED);
    motor2.write(MOTOR_SPEED);
    delay(1000);

    // Return to starting position
    motor1.write(-MOTOR_SPEED);
    motor2.write(-MOTOR_SPEED);
    delay(1000);
  }
}
